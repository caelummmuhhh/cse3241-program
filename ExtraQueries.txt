SELECT DRONES.Model, AVG(DRONE_MAINTENANCE.Cost) AS AverageMaintenanceCost
FROM DRONE_MAINTENANCE
JOIN DRONES ON DRONE_MAINTENANCE.DroneID = DRONES.DroneID
GROUP BY DRONES.Model;

SELECT WAREHOUSE.WarehouseID, SUM(VolumeCapacity) AS TotalDroneCapacity
FROM WAREHOUSE
JOIN INVENTORY ON WAREHOUSE.WarehouseID = INVENTORY.WarehouseID
JOIN DRONES ON INVENTORY.ItemID = DRONES.DroneID
GROUP BY WAREHOUSE.WarehouseID;

SELECT R.ReviewID, R.Rating, R.Comments, M.MemberID, M.FName, M.LName
FROM RENTAL_REVIEWS R
JOIN MEMBERS M ON R.MemberID = M.MemberID
WHERE R.Rating > 4;
